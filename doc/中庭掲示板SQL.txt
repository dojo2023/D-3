～ユーザー情報～
CREATE TABLE USER_INFO (
USER_EN VARCHAR (20),
USER_NAME VARCHAR (30),
USER_ID VARCHAR (30) PRIMARY KEY,
USER_PW VARCHAR (15),
USER_SQ_ID VARCHAR (30),
USER_SA VARCHAR (30),
USER_MODE_SWITCH INT (1),
CATEGORY_ID VARCHAR (30),
HASHTAGS_ID VARCHAR (30),
FREE_WORD VARCHAR (30),
FAVORITE_SWITCH INT (1)
); 

～秘密の質問～
CREATE TABLE USER_SQ (
USER_SQ_ID VARCHAR (30) PRIMARY KEY,
USER_SQ_NAME VARCHAR (50)
);

～投稿～
CREATE TABLE POSTER(
POSTER_ID VARCHAR (10) PRIMARY KEY,
TITLE VARCHAR (50),
CATEGORY_ID VARCHAR (30),
MAIN_SENTENCE VARCHAR (500),
HASHTAGS_ID VARCHAR (30),
POSTED_DATE VARCHAR (20),
ANIMAL_ID VARCHAR (30),
USER_ID VARCHAR (30),
USER_NAME_SWITCH INT (1)
);

～返信～
CREATE TABLE REPLY(
REPLY_ID VARCHAR (10) PRIMARY KEY,
POSTER_ID VARCHAR (10),
REPLY_SENTENCE VARCHAR (500),
REPLIED_DATE VARCHAR (20),
USER_NAME_SWITCH INT (1),
USER_ID VARCHAR (30),
ANIMAL_ID VARCHAR (30)
);

～通報～
CREATE TABLE REPORT(
REPORT_ID VARCHAR (10) PRIMARY KEY,
REPLY_ID VARCHAR (10),
POSTER_ID VARCHAR (10)
);

～カテゴリ～
CREATE TABLE CATEGORY(
CATEGORY_ID VARCHAR (30) PRIMARY KEY,
CATEGORY_NAME VARCHAR (20)
);

～ハッシュタグ～
CREATE TABLE HASHTAGS(
HASHTAGS_ID VARCHAR (30) PRIMARY KEY,
HASHTAGS_NAME VARCHAR (50)
);

～動物（匿名時）～
CREATE TABLE ANIMAL(
ANIMAL_ID VARCHAR (30) PRIMARY KEY,
ANIMAL_NAME VARCHAR (50)
);
